# -*- coding: utf-8 -*- """ 海南麻将核心逻辑类 适配规则：144张牌、有番条件、包牌/包杠、花杠、指定算分规则 """ import random from enum import Enum from typing import List, Dict, Tuple, Optional # ---------------------- 基础常量定义 ---------------------- # 牌型定义 SUITS = ['万', '筒', '条'] # 花色 NUMBERS = [str(i) for i in range(1, 10)] # 数字 HONORS = ['东', '南', '西', '北', '中', '发', '白'] # 字牌 FLOWERS = ['春', '夏', '秋', '冬', '梅', '兰', '竹', '菊'] # 花牌 ALL_FLOWERS = FLOWERS # 全套花牌 ALL_HONORS = HONORS + FLOWERS # 字牌+花牌 # 番型定义（对应规则里的有番条件） FAN_TYPES = { "四个顺子带一对": {"condition": lambda x: len(x["shunzi"])>=4 and len(x["pair"])==1, "multi": 1}, "门前清": {"condition": lambda x: x["chi"]==0 and x["peng"]==0 and x["angang"]==0, "multi": 1}, "二五八做将": {"condition": lambda x: x["pair_card"] in [str(i) for i in [2,5,8]] + SUITS, "multi": 1}, "中发白刻子/杠": {"condition": lambda x: any([len(c)>=3 for c in x["kezi"] if c[0] in ['中','发','白']]), "multi": 1}, "风位刻子/杠": {"condition": lambda x, seat: any([len(c)>=3 for c in x["kezi"] if c[0] == seat]), "multi": 1}, "令牌刻子": {"condition": lambda x: False, "multi": 1}, # 需补充令牌定义 "混一色": {"condition": lambda x: len(x["suits"])==1 and all([c[0] not in ALL_HONORS for c in x["tiles"]]), "multi": 2}, "抢杠胡": {"condition": lambda x: x["gang_hu"] is True, "multi": 1}, "自摸": {"condition": lambda x: x["self_draw"] is True, "multi": 1}, "碰碰胡": {"condition": lambda x: len(x["shunzi"])==0 and len(x["pair"])==1, "multi": 2}, "杠牌": {"condition": lambda x: len(x["gangs"])>0, "multi": 1} } # 牌型倍数（算分规则） HAND_MULTIPLIER = { "平胡": 1, "碰碰胡": 2, "混一色": 2, "清一色": 3, "七对": 2, "十三幺": 13 } # 结算常量 BASE_SCORE = { "花杠后胡": 3, "点炮": 1, "明杠": 1, "暗杠": 2, "首轮跟牌": 1 } # ---------------------- 工具类 ---------------------- class TileType(Enum): """牌的类型枚举""" WAN = 0 TONG = 1 TIAO = 2 HONOR = 3 # 东南西北中发白 FLOWER = 4 # 春夏秋冬梅兰竹菊 class HainanMahjongGame: """海南麻将核心游戏类""" def __init__(self, seat: str): """ 初始化游戏 :param seat: 玩家座位（东/南/西/北） """ self.seat = seat # 玩家座位（决定风位刻子条件） self.tiles = [] # 玩家手牌 self.shunzi = [] # 顺子列表 [['1万','2万','3万'], ...] self.kezi = [] # 刻子列表 [['1万','1万','1万'], ...] self.pair = [] # 对子 [['5万','5万']] self.pair_card = None # 对子牌面 self.gangs = [] # 杠牌列表 self.angang = 0 # 暗杠数量 self.minggang = 0 # 明杠数量 self.chi = 0 # 吃牌次数 self.peng = 0 # 碰牌次数 self.self_draw = False # 是否自摸胡 self.gang_hu = False # 是否抢杠胡 self.hu_type = None # 胡牌类型（平胡/碰碰胡等） self.fan_count = 0 # 番数 self.total_score = 0 # 最终得分 self.tiles_remaining = 144 # 剩余牌数 self.is_menqianqing = False # 是否门前清 # ---------------------- 核心功能：洗牌发牌 ---------------------- def shuffle_and_deal(self) -> List[str]: """洗牌并生成初始手牌（13张）""" # 生成144张牌 all_tiles = [] # 万筒条各36张 for suit in SUITS: for num in NUMBERS: all_tiles.extend([f"{num}{suit}"] * 4) # 字牌各4张 for honor in HONORS: all_tiles.extend([honor] * 4) # 花牌各1张（共8张） all_tiles.extend(FLOWERS) # 洗牌 random.shuffle(all_tiles) # 发牌（每人13张，庄家14张） if self.seat == '东': # 东为首局随机庄 self.tiles = all_tiles[:14] remaining = all_tiles[14:] else: self.tiles = all_tiles[:13] remaining = all_tiles[13:] self.tiles_remaining = len(remaining) return self.tiles # ---------------------- 核心功能：牌型拆解 ---------------------- def parse_tiles(self) -> None: """拆解手牌为顺子、刻子、对子""" # 简化版牌型拆解（实际需更复杂，适配核心规则） tile_dict = {} for tile in self.tiles: tile_dict[tile] = tile_dict.get(tile, 0) + 1 shunzi = [] kezi = [] pair = [] # 拆解顺子/刻子/对子 for tile, count in tile_dict.items(): if tile in FLOWERS: # 花牌不参与牌型拆解 continue if count == 2: pair.append([tile, tile]) self.pair_card = tile elif count == 3: kezi.append([tile, tile, tile]) elif count == 4: kezi.append([tile, tile, tile]) self.gangs.append([tile, tile, tile, tile]) self.minggang += 1 # 明杠统计 # 拆解顺子（万筒条内部） for suit in SUITS: suit_tiles = [t for t in self.tiles if t.endswith(suit) and t not in FLOWERS] suit_nums = sorted([int(t[:-1]) for t in suit_tiles]) i = 0 while i < len(suit_nums) - 2: if suit_nums[i] + 1 == suit_nums[i+1] and suit_nums[i+1] + 1 == suit_nums[i+2]: shunzi.append([f"{suit_nums[i]}{suit}", f"{suit_nums[i+1]}{suit}", f"{suit_nums[i+2]}{suit}"]) i += 3 else: i += 1 self.shunzi = shunzi self.kezi = kezi self.pair = pair # 判定门前清 self.is_menqianqing = self.chi == 0 and self.peng == 0 and self.angang == 0 # ---------------------- 核心功能：番型判断（有番条件） ---------------------- def check_fan_conditions(self) -> bool: """检查是否满足胡牌番型条件（必须满足至少一种）""" fan_conditions = [ # 1. 四个顺子带一对 (len(self.shunzi) >= 4 and len(self.pair) == 1), # 2. 门前清 self.is_menqianqing, # 3. 二五八做将 (self.pair_card and self.pair_card[:-1] in [str(i) for i in [2,5,8]]), # 4. 中发白刻子/杠 any([k[0] in ['中','发','白'] for k in self.kezi]), # 5. 自身风位刻子/杠 any([k[0] == self.seat for k in self.kezi]), # 6. 混一色 (len(set([t[-1] for t in self.tiles if t[-1] in SUITS])) == 1 and not any([t in ALL_HONORS for t in self.tiles])), # 7. 碰碰胡 (len(self.shunzi) == 0 and len(self.pair) == 1), # 8. 抢杠胡/自摸/杠牌 self.gang_hu or self.self_draw or (len(self.gangs) > 0) ] self.fan_count = sum(fan_conditions) return any(fan_conditions) # ---------------------- 核心功能：算分逻辑 ---------------------- def calculate_score(self, hu_type: str, is_dianpao: bool, gang_count: int, flower_count: int) -> int: """ 按规则计算最终得分 :param hu_type: 胡牌类型（平胡/碰碰胡/混一色/清一色/七对/十三幺） :param is_dianpao: 是否点炮胡 :param gang_count: 杠总数（明杠+暗杠） :param flower_count: 花牌数量 :return: 最终得分 """ # 基础分 base_multi = HAND_MULTIPLIER.get(hu_type, 1) base_score = BASE_SCORE.get("花杠后胡", 3) if flower_count > 0 else 1 # 额外分 extra_score = 0 if is_dianpao: extra_score += BASE_SCORE["点炮"] # 明杠/暗杠分 extra_score += self.minggang * BASE_SCORE["明杠"] extra_score += self.angang * BASE_SCORE["暗杠"] # 首轮跟牌分（需记录首轮出牌） # extra_score += BASE_SCORE["首轮跟牌"] # 需补充出牌逻辑 # 花杠分 if flower_count >= 7: extra_score += 1 if flower_count >= 4: extra_score += flower_count // 4 # 凑4种花加1杠分 # 总得分公式：输赢分数=基础分*牌型倍数*(噶数+1+庄分)+额外分 # 简化版（适配核心逻辑） total_score = base_score * base_multi + extra_score self.total_score = total_score return total_score # ---------------------- 核心功能：包牌/包杠规则 ---------------------- def check_bao_pai(self, a_seat: str, b_seat: str, chi_count: int, peng_count: int) -> Dict: """ 检查包牌规则 :param a_seat: 玩家A座位 :param b_seat: 玩家B座位 :param chi_count: A吃B的次数 :param peng_count: A碰B的次数 :return: 包牌结果 """ bao_result = {"is_bao": False, "bearer": None, "rule": None} total_chi_peng = chi_count + peng_count # 三道包牌：吃碰3次 if total_chi_peng == 3: bao_result = {"is_bao": True, "bearer": f"{a_seat}与{b_seat}互包", "rule": "三道包牌"} # 四道包牌：吃碰4次 elif total_chi_peng == 4: bao_result = {"is_bao": True, "bearer": f"{a_seat}与{b_seat}互包（含自摸）", "rule": "四道包牌"} return bao_result def check_bao_gang(self, first_discard: str, gang_tile: str) -> Dict: """ 检查包杠规则 :param first_discard: 玩家首轮出牌 :param gang_tile: 杠的牌 :return: 包杠结果 """ bao_result = {"is_bao": False, "bearer": None, "rule": None} # 玩家首轮出的牌点的明杠需要包此杠 if first_discard == gang_tile and len(self.gangs) > 0: bao_result = {"is_bao": True, "bearer": "首轮出牌玩家包杠", "rule": "包杠规则"} return bao_result # ---------------------- 核心功能：花杠规则 ---------------------- def calculate_flower_gang(self, flower_tiles: List[str], is_choose_flowers: bool) -> Tuple[int, int]: """ 计算花杠分 :param flower_tiles: 花牌列表 :param is_choose_flowers: 是否勾选花牌规则 :return: (杠分数, 额外分) """ flower_count = len(flower_tiles) gang_score = 0 extra_score = 0 if is_choose_flowers: # 勾选：春夏秋冬/梅兰竹菊算花杠，摸到8张加3杠分 gang_score = flower_count // 8 * 3 + flower_count // 4 else: # 不勾选：摸到7张算花杠，摸到8张加3杠分 if flower_count >= 7: gang_score = 1 if flower_count >= 8: gang_score += 3 # 凑齐4种花加1杠分，凑齐7种加1分 unique_flowers = len(set(flower_tiles)) if unique_flowers >= 4: extra_score += 1 if unique_flowers >= 7: extra_score += 1 return gang_score, extra_score
